    public SqQueue(int maxSize) {

            maxSize = 1;

        this.maxSize = maxSize;
        this.front = 0;
        this.rear = 0;
        this.datas = (T[]) new Object[this.maxSize];
    }

    //ä¸¤ä¸ªçŠ¶æ€:é˜Ÿç©º&é˜Ÿæ»¡
    public boolean isNull() {
        if (this.front == this.rear)
            return true;
        else
            return false;
    }

    public boolean isFull() {
        if ((rear + 1) % this.maxSize == front)
            return true;
        else
            return false;
    }

    //åˆå§‹åŒ–é˜Ÿåˆ—
    public void initQueue() {
        this.front = 0;
        this.front = 0;
    }

    //ä¸¤ä¸ªæ“ä½œ:è¿›é˜Ÿ&å‡ºé˜Ÿ
    public boolean push(T data) {
        if (isFull())
            return false;//é˜Ÿæ»¡åˆ™æ— æ³•è¿›é˜Ÿ
        else {
            datas[rear] = data;//è¿›é˜Ÿ
            rear = (rear + 1) % maxSize;//é˜Ÿå°¾æŒ‡é’ˆ+1.
            return true;
        }
    }

        else {
            T popData = datas[front++];//å‡ºé˜Ÿ

            return popData;
        }
    }

    }

    }

    }

    }

        this.maxSize = 0;
    }

    //åˆå§‹åŒ–é˜Ÿåˆ—

        front.next = null;
        rear.next = null;
        maxSize = 0;
    }

    //é˜Ÿç©ºåˆ¤æ–­
    public boolean isNull() {
        if (front.next == null || rear.next == null)
            return true;
        else
            return false;
    }

    //è¿›é˜Ÿ
    public void push(QNode<T> node) {
        if (isNull()) {
            //ç¬¬ä¸€æ¬¡
            front.next = node;
            rear.next = node;
            maxSize++;
        } else {
            node.next = front.next;
            front.next = node;
            maxSize++;
        }
    }

    //å‡ºé˜Ÿ
    public QNode<T> pop() {
        if (isNull())
            return null;//é˜Ÿä¸ºç©ºæ—¶ï¼Œæ— æ³•å‡ºé˜Ÿ
