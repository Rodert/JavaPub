List

Mr.WangеӨ§зәҰ 4 еҲҶй’ҹгҖҠйқўиҜ•1v1гҖӢ

жҲ‘жҳҜ javapubпјҢдёҖеҗҚ Markdown зЁӢеәҸе‘ҳд»Һрҹ‘ЁвҖҚрҹ’»пјҢе…«иӮЎж–Үз§ҚеӯҗйҖүжүӢгҖӮ

пјҡ е°Ҹдјҷеӯҗ,еҗ¬иҜҙдҪ еҜ№JavaйӣҶеҗҲжҢәеңЁиЎҢзҡ„?

и°ўи°ўеӨёеҘ–,жҲ‘еҜ№JavaйӣҶеҗҲиҝҳеңЁеӯҰд№ дёӯ,еҸӘиғҪз®—е…Ҙй—Ёж°ҙе№ігҖӮзү№еҲ«жҳҜListиҝҷдёӘжҺҘеҸЈ,е…¶дёӢзҡ„е®һзҺ°зұ»еҠҹиғҪйқһеёёдё°еҜҢ,жҲ‘иҝҳжңӘиғҪе…ЁйғЁжҺҢжҸЎгҖӮ

пјҡ йӮЈд№Ҳ,з®ҖеҚ•д»Ӣз»ҚдёӢListиҝҷдёӘжҺҘеҸЈеҸҠеёёз”Ёе®һзҺ°зұ»еҗ§!иҝҷжҳҜJavaйӣҶеҗҲзҡ„еҹәзЎҖ,д№ҹжҳҜж—ҘеёёејҖеҸ‘дёӯжңҖеёёз”Ёзҡ„гҖӮ

ListжҺҘеҸЈиЎЁзӨәдёҖдёӘжңүеәҸйӣҶеҗҲ,е®ғзҡ„дё»иҰҒе®һзҺ°зұ»жңү:

  1. ArrayList:еҹәдәҺеҠЁжҖҒж•°з»„е®һзҺ°,жҹҘиҜўеҝ«гҖҒеўһеҲ ж…ўгҖӮ
ArrayList<Integer> list = new ArrayList<>();
  1. LinkedList:еҹәдәҺеҸҢеҗ‘й“ҫиЎЁе®һзҺ°,жҹҘиҜўж…ўгҖҒеўһеҲ еҝ«гҖӮ
LinkedList<Integer> list = new LinkedList<>();
  1. Vector:е’ҢArrayListзұ»дјј,дҪҶе®ғжҳҜзәҝзЁӢе®үе…Ёзҡ„,жҹҘиҜўеҝ«гҖҒеўһеҲ ж…ўгҖӮ
Vector<Integer> list = new Vector<>();
  1. Stack:з»§жүҝVector,е®һзҺ°ж Ҳз»“жһ„,еҗҺиҝӣе…ҲеҮәгҖӮ
Stack<Integer> stack = new Stack<>();

е®ғд»¬йғҪе®һзҺ°дәҶListжҺҘеҸЈ,жңүдёҖдәӣе…ұеҗҢзҡ„ж–№жі•,дҪҶеә•еұӮж•°жҚ®з»“жһ„дёҚеҗҢ,жүҖд»ҘеңЁдёҚеҗҢеңәжҷҜжңүдёҚеҗҢзҡ„дҪҝз”ЁдјҳеҠҝгҖӮиҝҷеҸ–еҶідәҺеә”з”Ёзҡ„йңҖжұӮгҖӮ

пјҡ йӮЈж—Ҙеёёе·ҘдҪңз”Ёзҡ„жңҖеӨҡзҡ„жҳҜе“ӘдёӘе®һзҺ°зұ»?е®ғзҡ„жәҗз ҒиғҪдёҚиғҪи®Іи§ЈдёҖдёӢ?

жҲ‘ж—Ҙеёёе·ҘдҪңдёӯжңҖеёёз”Ёзҡ„Listе®һзҺ°зұ»е°ұжҳҜArrayListгҖӮ е®ғзҡ„жәҗз ҒеҰӮдёӢ:

public class ArrayList<E> extends AbstractList<E> 
        implements List<E>, RandomAccess, Cloneable, java.io.Serializable
{
    /**
     * й»ҳи®ӨеҲқе§Ӣе®№йҮҸ
     */
    private static final int DEFAULT_CAPACITY = 10;

    /**
     * еӯҳеӮЁе…ғзҙ зҡ„ж•°з»„
     */
    transient Object[] elementData; 

    /**
     * е®һйҷ…е®№йҮҸ  
     */
    private int size;

    public ArrayList(int initialCapacity) {
        if (initialCapacity > 0) {
            this.elementData = new Object[initialCapacity];
        } else {
            this.elementData = new Object[DEFAULT_CAPACITY];
        }
    }

    public ArrayList() {
        this.elementData = new Object[DEFAULT_CAPACITY];
    }
    
    //ж·»еҠ е…ғзҙ 
    public boolean add(E e) {
        ensureCapacityInternal(size + 1);   // Increments modCount!!
        elementData[size++] = e;
        return true;
    }  
}

ArrayListеә•еұӮйҮҮз”ЁеҠЁжҖҒж•°з»„е®һзҺ°,йҖҡиҝҮensureCapacityInternal()ж–№жі•еҠЁжҖҒжү©е®№,д»ҘиҫҫеҲ°еңЁдҝқиҜҒжҹҘиҜўж•ҲзҺҮзҡ„еҗҢж—¶,е°ҪйҮҸеҮҸе°Ҹжү©е®№еёҰжқҘзҡ„жҖ§иғҪж¶ҲиҖ—гҖӮиҝҷд№ҹжҳҜжҲ‘еңЁж—ҘеёёдҪҝз”ЁдёӯжңҖж¬ЈиөҸArrayListзҡ„ең°ж–№гҖӮеҪ“з„¶,е®ғзҡ„е®һзҺ°иҝңдёҚжӯўиҝҷдәӣ,жҲ‘иҝҳеңЁдёҚж–ӯеӯҰд№ дёҺзҗҶи§ЈдёӯгҖӮ

пјҡ дёҚй”ҷ,дҪ еҜ№иҝҷдәӣзҹҘиҜҶе·Із»ҸжңүдёҖе®ҡзҗҶи§ЈгҖӮArrayListзҡ„жәҗз ҒеҲҶжһҗеҫ—д№ҹжҜ”иҫғеҲ°дҪҚгҖӮзңӢжқҘдҪ д№ӢеүҚзңҹзҡ„жңүи®Өзңҹз ”иҜ»дёҺзҗҶи§ЈгҖӮдёҚиҝҮListзӣёе…ізҹҘиҜҶиҝҳжңүжӣҙе№ҝйҳ”зҡ„з©әй—ҙ,йңҖиҰҒдҪ з»§з»ӯеҠӘеҠӣ!

йқһеёёж„ҹи°ўйқўиҜ•е®ҳзҡ„иӮҜе®ҡдёҺжҢҮеҜјгҖӮжӮЁиҜҙеҫ—еҜ№,ListеҸҠе…¶зӣёе…ізҹҘиҜҶиҝҳжңүеҫҲеӨҡеҖјеҫ—жҲ‘з»§з»ӯеӯҰд№ дёҺжҺўзҙўзҡ„ең°ж–№гҖӮжҲ‘дјҡжҢҒз»ӯеҠ ж·ұзҗҶи§Ј,жҸҗй«ҳиҝҗз”ЁиғҪеҠӣгҖӮ

пјҡ йӮЈд№Ҳ,дҪ еҜ№Listиҝҳжңүе“ӘдәӣдёҚеӨӘзҗҶи§Јзҡ„ең°ж–№?жҲ–жҳҜжғіжӣҙж·ұе…ҘеӯҰд№ зҡ„еҶ…е®№?

е…ідәҺList,жҲ‘иҝҳдёҚеӨӘжё…жҘҡжҲ–жғіиҝӣдёҖжӯҘеӯҰд№ зҡ„еҶ…е®№еҰӮдёӢ:

  1. CopyOnWriteArrayList:е®ғзҡ„е®һзҺ°еҺҹзҗҶдёҺзәҝзЁӢе®үе…ЁжңәеҲ¶,иҝҷд№ҹжҳҜж—ҘеёёејҖеҸ‘дёӯз»ҸеёёдҪҝз”Ёзҡ„е№¶еҸ‘ListгҖӮ
  2. SubList :ListеӯҗеҲ—иЎЁзҡ„е®һзҺ°еҺҹзҗҶдёҺиҫ№з•Ңжғ…еҶөгҖӮд»ҘеҸҠеӯҗеҲ—иЎЁзҡ„дҝ®ж”№еҰӮдҪ•еҪұе“ҚеҺҹеҲ—иЎЁгҖӮ
  3. Listеҗ„е®һзҺ°зұ»еңЁе№¶еҸ‘зҺҜеўғдёӢзҡ„иЎЁзҺ°гҖӮеҰӮдҪ•йҖүжӢ©жҒ°еҪ“зҡ„е№¶еҸ‘Listзұ»?
  4. Listиҝӯд»ЈеҷЁ:ListIteratorдёҺIteratorзҡ„еҢәеҲ«дёҺдҪҝз”ЁеңәжҷҜгҖӮ
  5. QueueжҺҘеҸЈ:QueueдҪңдёәListзҡ„еӯҗжҺҘеҸЈ,е…¶е®һзҺ°зұ»дёҺдҪҝз”ЁеңәжҷҜгҖӮ
  6. еә•еұӮж•°жҚ®з»“жһ„:зҗҶи§Јж•°з»„гҖҒй“ҫиЎЁгҖҒзәўй»‘ж ‘зӯүеҜ№Listе®һзҺ°зҡ„еҪұе“Қ,д»ҘеҸҠеҗ„иҮӘзҡ„дјҳеҠЈгҖӮ
  7. жү©е®№жңәеҲ¶:дёҚеҗҢListзҡ„жү©е®№зӯ–з•ҘеҸҠжҖ§иғҪеҪұе“ҚгҖӮеҰӮдҪ•дјҳеҢ–?
  8. еәҸеҲ—еҢ–дёҺеҸҚеәҸеҲ—еҢ–:Listеҗ„е®һзҺ°зұ»зҡ„еәҸеҲ—еҢ–дёҺеҸҚеәҸеҲ—еҢ–е®һзҺ°гҖӮ
  9. з©әеҖјеӨ„зҗҶ:nullеҖјеңЁеҗ„Listе®һзҺ°зұ»дёӯзҡ„еӨ„зҗҶж–№ејҸгҖӮ

иҝҷдәӣйғҪжҳҜжҲ‘жғіиҝӣдёҖжӯҘеӯҰд№ дёҺзҗҶи§Јзҡ„Listзӣёе…іеҶ…е®№дёҺзҹҘиҜҶзӮ№гҖӮжҲ‘дјҡж №жҚ®иҝҷд»Ҫжё…еҚ•з»§з»ӯж·ұе…Ҙйҳ…иҜ»жәҗз ҒгҖҒеҲҶжһҗжЎҲдҫӢе№¶е®һи·өдҪҝз”Ё,д»Ҙдҫҝе…ЁйқўжҺҢжҸЎListеҸҠе…¶зӣёе…іжҺҘеҸЈдёҺе®һзҺ°зұ»гҖӮиҝҷж— з–‘йңҖиҰҒдёҖж®өй•ҝжңҹзҡ„еӯҰд№ дёҺжҖ»з»“иҝҮзЁӢ,дҪҶиҝҷжӯЈжҳҜжҲ‘жҲҗй•ҝдёәдёҖеҗҚиө„ж·ұJavaе·ҘзЁӢеёҲжүҖеҝ…йЎ»з»ҸеҺҶзҡ„йҳ¶ж®өгҖӮ

пјҡ Wonderful! иҝҷд»ҪеӯҰд№ жё…еҚ•ж¶өзӣ–зҡ„еҶ…е®№йқһеёёе…Ёйқўдё”е…·жңүй’ҲеҜ№жҖ§гҖӮдҪ иғҪеӨҹеҮҶзЎ®е®ҡдҪҚиҮӘе·ұе°ҡжңӘе®Ңе…ЁжҺҢжҸЎзҡ„зҹҘиҜҶзӮ№,иҝҷеұ•зҺ°еҮәдҪ зҡ„иҮӘжҲ‘и®ӨзҹҘиғҪеҠӣгҖӮеҸӘиҰҒдҪ иғҪеӨҹжңүи®ЎеҲ’е’ҢиҖҗеҝғең°еҗ‘иҝҷдёӘжё…еҚ•дёҠзҡ„жҜҸдёҖйЎ№зҹҘиҜҶзӮ№иҝӣеҸ‘,дҪ еңЁListеҸҠзӣёе…іжҺҘеҸЈзҡ„зҗҶи§ЈдёҠдёҖе®ҡдјҡжңүеӨ§зҡ„жҸҗй«ҳ,иҝҷд№ҹдёәдҪ жҲҗй•ҝдёәиө„ж·ұе·ҘзЁӢеёҲеҘ е®ҡеҹәзЎҖгҖӮжҲ‘еҜ№дҪ зҡ„еӯҰд№ жҖҒеәҰе’ҢзҗҶи§ЈиғҪеҠӣеҫҲдёәж¬ЈиөҸгҖӮ

жңҖиҝ‘жҲ‘еңЁжӣҙж–°гҖҠйқўиҜ•1v1гҖӢзі»еҲ—ж–Үз« пјҢдё»иҰҒд»ҘеңәжҷҜеҢ–зҡ„ж–№ејҸпјҢи®Іи§ЈжҲ‘д»¬еңЁйқўиҜ•дёӯйҒҮеҲ°зҡ„й—®йўҳпјҢиҮҙеҠӣдәҺи®©жҜҸдёҖдҪҚе·ҘзЁӢеёҲжӢҝеҲ°иҮӘе·ұеҝғд»Әзҡ„offerпјҢж„ҹе…ҙи¶ЈеҸҜд»Ҙе…іжіЁе…¬дј—еҸ·JavaPubиҝҪжӣҙпјҒ
